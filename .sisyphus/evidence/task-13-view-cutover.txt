SECTION 5: NEW VIEW SQL

```sql
CREATE OR REPLACE VIEW campsite_full AS
SELECT
    -- Identity
    p.id::text AS place_id,
    COALESCE(up.name, op.name, gp.name, lp.name) AS name,
    COALESCE(
        NULLIF(TRIM(CONCAT_WS(', ',
            COALESCE(up.city, op.city, gp.city, lp.city),
            COALESCE(up.region, op.region, gp.region, lp.region),
            COALESCE(up.country_code, op.country_code, gp.country_code, lp.country_code)
        )), ''),
        COALESCE(up.address, op.address, gp.address, lp.address),
        COALESCE(up.name, op.name, gp.name, lp.name)
    ) AS location,
    p.lat,
    p.lon AS lng,

    -- Source / Type
    COALESCE(up.source_place_type, op.source_place_type, gp.source_place_type, lp.source_place_type)::text AS source_type,
    op.osm_id::text AS osm_id,
    jsonb_build_object(
        'source_place_type', COALESCE(up.source_place_type, op.source_place_type, gp.source_place_type, lp.source_place_type),
        'place_type', COALESCE(up.place_type, op.place_type, gp.place_type, lp.place_type),
        'country_code', COALESCE(up.country_code, op.country_code, gp.country_code, lp.country_code),
        'city', COALESCE(up.city, op.city, gp.city, lp.city)
    ) AS osm_tags,

    -- Contact / Operations
    COALESCE(up.website, op.website, gp.website, lp.website) AS website,
    COALESCE(up.opening_hours, op.opening_hours, gp.opening_hours, lp.opening_hours) AS opening_hours,
    COALESCE(up.phone, op.phone, gp.phone, lp.phone) AS contact_phone,
    COALESCE(up.email, op.email, gp.email, lp.email) AS contact_email,

    -- Scraped / Cache data (unchanged from campsites_cache)
    cc.scraped_website_url,
    cc.scraped_at,
    cc.scraped_price_info,
    cc.scraped_data_source::text AS scraped_data_source,
    cc.description,
    cc.description_source::text AS description_source,
    cc.description_generated_at,
    cc.description_version,
    cc.estimated_price::numeric AS estimated_price,
    cc.price_source::text AS price_source,
    cc.user_price_count::bigint AS user_price_count,
    cc.user_price_avg::numeric AS user_price_avg,
    COALESCE(cc.place_types, ARRAY[COALESCE(up.place_type, op.place_type, gp.place_type, lp.place_type)]::text[]) AS place_types,
    p.created_at,
    p.updated_at,

    -- Google data (unchanged from campsites_cache)
    cc.place_id AS google_place_id,
    cc.google_photos,
    cc.google_reviews,
    cc.google_data_fetched_at,
    cc.google_data_expires_at,
    CASE
        WHEN cc.google_data_expires_at IS NOT NULL AND cc.google_data_expires_at < NOW()
        THEN true
        ELSE false
    END AS google_data_expired,

    -- Aggregates (unchanged)
    COALESCE(r.review_count, 0) AS review_count,
    r.avg_rating,
    COALESCE(cp.price_count, 0) AS user_price_entries,
    COALESCE(f.favorite_count, 0) AS favorite_count,
    cc.rating::numeric AS google_rating,
    COALESCE(cc.rating::numeric, r.avg_rating) AS rating,

    -- Type derivation
    CASE
        WHEN COALESCE(up.place_type, op.place_type, gp.place_type, lp.place_type)::text = 'camp_site' THEN 'camping'
        WHEN COALESCE(up.place_type, op.place_type, gp.place_type, lp.place_type)::text = 'camper_stop' THEN 'stellplatz'
        ELSE 'poi'
    END AS type,

    -- Amenities (resolved from property tables with column name mapping)
    COALESCE(up.has_restrooms, op.has_restrooms, gp.has_restrooms, lp.has_restrooms, false) AS has_toilet,
    COALESCE(up.has_shower, op.has_shower, gp.has_shower, lp.has_shower, false) AS has_shower,
    COALESCE(up.has_electricity, op.has_electricity, gp.has_electricity, lp.has_electricity, false) AS has_electricity,
    COALESCE(up.pets_allowed, op.pets_allowed, gp.pets_allowed, lp.pets_allowed, false) AS has_dogs_allowed,
    COALESCE(up.has_wifi, op.has_wifi, gp.has_wifi, lp.has_wifi, false) AS has_wifi,
    COALESCE(up.has_beach, op.has_beach, gp.has_beach, lp.has_beach, false) AS has_beach,
    COALESCE(up.has_laundry, op.has_laundry, gp.has_laundry, lp.has_laundry, false) AS has_laundry,
    COALESCE(up.has_restaurant, op.has_restaurant, gp.has_restaurant, lp.has_restaurant, false) AS has_restaurant,
    COALESCE(up.has_cafe, op.has_cafe, gp.has_cafe, lp.has_cafe, false) AS has_bar,
    COALESCE(up.has_shop, op.has_shop, gp.has_shop, lp.has_shop, false) AS has_shop,
    COALESCE(up.has_pool, op.has_pool, gp.has_pool, lp.has_pool, false) AS has_pool,
    COALESCE(up.has_playground, op.has_playground, gp.has_playground, lp.has_playground, false) AS has_playground,
    COALESCE(up.has_dump_station, op.has_dump_station, gp.has_dump_station, lp.has_dump_station, false) AS has_dump_station,
    COALESCE(up.has_drinking_water, op.has_drinking_water, gp.has_drinking_water, lp.has_drinking_water, false) AS has_water,
    COALESCE(up.has_laundry, op.has_laundry, gp.has_laundry, lp.has_laundry, false) AS has_washing_machine,
    COALESCE(up.has_dishwashing_area, op.has_dishwashing_area, gp.has_dishwashing_area, lp.has_dishwashing_area, false) AS has_dishwasher
FROM places p
LEFT JOIN place_user_properties up ON up.place_id = p.id AND up.is_current = true
LEFT JOIN place_osm_properties op ON op.place_id = p.id AND op.is_current = true
LEFT JOIN place_google_properties gp ON gp.place_id = p.id AND gp.is_current = true
LEFT JOIN place_llm_properties lp ON lp.place_id = p.id AND lp.is_current = true
LEFT JOIN campsites_cache cc ON cc.place_id = p.id::text
LEFT JOIN (
    SELECT place_id, COUNT(*) AS review_count, AVG(rating) AS avg_rating
    FROM campsite_reviews
    GROUP BY place_id
) r ON r.place_id = p.id::text
LEFT JOIN (
    SELECT place_id, COUNT(*) AS price_count
    FROM campsite_prices
    GROUP BY place_id
) cp ON cp.place_id = p.id::text
LEFT JOIN (
    SELECT place_id, COUNT(*) AS favorite_count
    FROM favorites
    GROUP BY place_id
) f ON f.place_id = p.id::text;
```
